	ALTER ROLE Usario_DB SET default_transaction_isolation TO 'read committed';
	ALTER ROLE Usario_DB SET timezone TO 'UTC'
	
CAMBIO BASE DB	
	\c db_practica_orm;
GRANT ALL PRIVILEGES ON SCHEMA public TO user_db;
ALTER ROLE user_db SET search_path TO public;
